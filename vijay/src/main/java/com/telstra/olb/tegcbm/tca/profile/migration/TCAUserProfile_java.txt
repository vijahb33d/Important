package com.telstra.olb.tegcbm.tca.profile.migration;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

/**
 * 
 * Java POJO class used to store the TCA profile attributes
 */
public class TCAUserProfile {

    private static final String QUOTE = "\"";

    private static final String COMMA = ",";

    private String cidn;

    private String ecsid;

    private List attribs = new ArrayList();

    private Map attributes = new HashMap();

    /**
     * @return Returns the cidn.
     */
    public String getCidn() {
        return cidn;
    }

    /**
     * @param cidn
     *            The cidn to set.
     */
    public void setCidn(String cidn) {
        this.cidn = cidn;
    }

    /**
     * @return Returns the ecsid.
     */
    public String getEcsid() {
        return ecsid;
    }

    /**
     * @param ecsid
     *            The ecsid to set.
     */
    public void setEcsid(String ecsid) {
        this.ecsid = ecsid;
    }

    public String toString() {
        StringBuffer sb = new StringBuffer();
        sb.append(QUOTE + this.getCidn() + QUOTE + COMMA);
        sb.append(QUOTE + this.getEcsid() + QUOTE);
        for (int index = 0; index < attribs.size(); index++) {
            sb.append(COMMA + QUOTE + this.attributes.get(attribs.get(index)) + QUOTE);
        }
        return sb.toString();
    }

    /**
     * @return Returns the attribs.
     */
    public List getAttribs() {
        return attribs;
    }

    /**
     * @param attribs
     *            The attribs to set.
     */
    public void setAttribs(List attribs) {
        this.attribs = attribs;
    }

    /**
     * @return Returns the attributes.
     */
    public Map getAttributes() {
        return attributes;
    }

    /**
     * @param attributes
     *            The attributes to set.
     */
    public void setAttribute(String key, String value) {
        this.attributes.put(key, value);
    }

}
