/**
 * Populate Data Exception.
 */
public class TBRegnException extends Exception {
    
    /**
     * 
     */
    public TBRegnException() {
        super();

    }
    /**
     * @param message exception message.
     */
    public TBRegnException(String message) {
        super(message);

    }

        super(message, cause);

    }
    /**
     * @param cause exception cause.
     */
    public TBRegnException(Throwable cause) {
        super(cause);

    }
}
